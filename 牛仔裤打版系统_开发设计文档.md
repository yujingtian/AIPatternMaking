class CurveType(Enum):
    """曲线类型枚举"""
    LINE = "LINE"
    BEZIER_CUBIC = "BEZIER_CUBIC"
    BEZIER_QUADRATIC = "BEZIER_QUADRATIC"
    SPLINE_CATMULL_ROM = "SPLINE_CATMULL_ROM"

    GRAIN = 4
    TEXT = 5
    STITCH = 6

@dataclass
class Point:
    """2D点"""
    x: float
    y: float
    name: Optional[str] = None

    def __add__(self, other: 'Point') -> 'Point':
        return Point(self.x + other.x, self.y + other.y)

    def __sub__(self, other: 'Point') -> 'Point':
        return Point(self.x - other.x, self.y - other.y)

    def __mul__(self, scalar: float) -> 'Point':
        return Point(self.x * scalar, self.y * scalar)

    def distance_to(self, other: 'Point') -> float:
        """计算到另一个点的距离"""
        return ((self.x - other.x)**2 + (self.y - other.y)**2)**0.5

    def to_tuple(self) -> Tuple[float, float]:
        return (self.x, self.y)

    def copy(self) -> 'Point':
        return Point(self.x, self.y, self.name)


@dataclass
class BezierCurve:
    """贝塞尔曲线（三次）"""
    p0: Point
    p1: Point  # 控制点1
    p2: Point  # 控制点2
    p3: Point
    type: CurveType = CurveType.BEZIER_CUBIC

    def point_at(self, t: float) -> Point:
        """获取参数t处的点，t ∈ [0,1]"""
        x = (1-t)**3 * self.p0.x + 3*(1-t)**2 * t * self.p1.x + \
            3*(1-t) * t**2 * self.p2.x + t**3 * self.p3.x
        y = (1-t)**3 * self.p0.y + 3*(1-t)**2 * t * self.p1.y + \
            3*(1-t) * t**2 * self.p2.y + t**3 * self.p3.y
        return Point(x, y)

    def length(self, segments: int = 100) -> float:
        """计算曲线长度（分段折线逼近）"""
        total = 0.0
        for i in range(segments):
            t1 = i / segments
            t2 = (i + 1) / segments
            p1 = self.point_at(t1)
            p2 = self.point_at(t2)
            total += p1.distance_to(p2)
        return total

    def to_polyline(self, segments: int = 100) -> List[Point]:
        """转换为折线"""
        return [self.point_at(i/segments) for i in range(segments + 1)]


@dataclass
class QuadraticBezierCurve:
    """贝塞尔曲线（二次）"""
    p0: Point
    p1: Point  # 控制点
    p2: Point
    type: CurveType = CurveType.BEZIER_QUADRATIC

    def point_at(self, t: float) -> Point:
        x = (1-t)**2 * self.p0.x + 2*(1-t)*t * self.p1.x + t**2 * self.p2.x
        y = (1-t)**2 * self.p0.y + 2*(1-t)*t * self.p1.y + t**2 * self.p2.y
        return Point(x, y)

    def length(self, segments: int = 100) -> float:
        total = 0.0
        for i in range(segments):
            t1 = i / segments
            t2 = (i + 1) / segments
            p1 = self.point_at(t1)
            p2 = self.point_at(t2)
            total += p1.distance_to(p2)
        return total

    def to_polyline(self, segments: int = 100) -> List[Point]:
        return [self.point_at(i/segments) for i in range(segments + 1)]


@dataclass
class LineSegment:
    """直线段"""
    p0: Point
    p1: Point
    type: CurveType = CurveType.LINE

    def length(self) -> float:
        return self.p0.distance_to(self.p1)

    def point_at(self, t: float) -> Point:
        """t ∈ [0,1]"""
        return Point(
            self.p0.x + (self.p1.x - self.p0.x) * t,
            self.p0.y + (self.p1.y - self.p0.y) * t
        )

    def to_polyline(self, segments: int = 2) -> List[Point]:
        return [self.p0, self.p1]


@dataclass
class CatmullRomSpline:
    """Catmull-Rom样条曲线"""
    points: List[Point]
    tension: float = 0.5
    type: CurveType = CurveType.SPLINE_CATMULL_ROM

    def point_at_segment(self, segment_idx: int, t: float) -> Point:
        """获取某一段上的点，segment_idx从0开始，t ∈ [0,1]"""
        if len(self.points) < 4:
            raise ValueError("Catmull-Rom样条至少需要4个点")

        p0 = self.points[max(0, segment_idx - 1)]
        p1 = self.points[segment_idx]
        p2 = self.points[min(len(self.points) - 1, segment_idx + 1)]
        p3 = self.points[min(len(self.points) - 1, segment_idx + 2)]

        t2 = t * t
        t3 = t2 * t

        x = 0.5 * (
            2 * p1.x +
            (-p0.x + p2.x) * t +
            (2*p0.x - 5*p1.x + 4*p2.x - p3.x) * t2 +
            (-p0.x + 3*p1.x - 3*p2.x + p3.x) * t3
        )
        y = 0.5 * (
            2 * p1.y +
            (-p0.y + p2.y) * t +
            (2*p0.y - 5*p1.y + 4*p2.y - p3.y) * t2 +
            (-p0.y + 3*p1.y - 3*p2.y + p3.y) * t3
        )
        return Point(x, y)

    def to_polyline(self, segments_per_knot: int = 20) -> List[Point]:
        """转换为折线"""
        result = []
        for i in range(len(self.points) - 1):
            for j in range(segments_per_knot):
                t = j / segments_per_knot
                result.append(self.point_at_segment(i, t))
        result.append(self.points[-1])
        return result

    def length(self, segments_per_knot: int = 20) -> float:
        polyline = self.to_polyline(segments_per_knot)
        total = 0.0
        for i in range(len(polyline) - 1):
            total += polyline[i].distance_to(polyline[i+1])
        return total

```python
# 文件: src/calculator.py
"""
中间变量计算器
按顺序计算所有中间变量
"""
from .models import (
    JeansPatternInput, IntermediateVariables
)


class IntermediateCalculator:
    """中间变量计算器"""

    def __init__(self, input_data: JeansPatternInput):
        self.input = input_data
        self.vars = IntermediateVariables()

    def calculate(self) -> IntermediateVariables:
        """执行所有计算步骤"""
        self._step1_net_dimensions()
        self._step2_horizontal_lines()
        self._step3_crotch_width()
        self._step4_thigh_width()
        self._step5_crease_line()
        self._step6_waist_distribution()
        self._step7_knee_ankle()
        return self.vars

    def _step1_net_dimensions(self):
        """步骤1: 成品尺寸计算"""
        m = self.input.measurements
        e = self.input.ease
        self.vars.Net_W = m.W + e.Ease_W
        self.vars.Net_H = m.H + e.Ease_H
        self.vars.Net_Knee = m.Knee + e.Ease_Knee
        self.vars.Net_Ankle = m.Ankle + e.Ease_Ankle

    def _step2_horizontal_lines(self):
        """步骤2: 水平线位置计算"""
        m = self.input.measurements
        d = self.input.design
        self.vars.Waist_Line_Y = 0.0
        self.vars.Thigh_Line_Y = m.Rise - d.Waistband_Width
        self.vars.Hip_Line_Y = self.vars.Thigh_Line_Y * d.Hip_Line_Ratio
        self.vars.Ankle_Line_Y = m.L - d.Waistband_Width
        self.vars.Knee_Line_Y = self.vars.Thigh_Line_Y + \
            (self.vars.Ankle_Line_Y - self.vars.Thigh_Line_Y) / 2 - 4.0

    def _step3_crotch_width(self):
        """步骤3: 裆宽计算"""
        self.vars.Front_Crotch_Width = self.vars.Net_H / 20 + 0.5
        self.vars.Back_Crotch_Width = self.vars.Net_H / 10

    def _step4_thigh_width(self):
        """步骤4: 横裆宽计算"""
        self.vars.Front_Thigh_Width = self.vars.Net_H / 4 - 1.5 + self.vars.Front_Crotch_Width
        self.vars.Back_Thigh_Width = self.vars.Net_H / 4 + 1.5 + self.vars.Back_Crotch_Width

    def _step5_crease_line(self):
        """步骤5: 烫迹线位置计算"""
        self.vars.Front_Crease_X = self.vars.Front_Thigh_Width / 2 - self.vars.Front_Crotch_Width
        self.vars.Back_Crease_X = self.vars.Back_Thigh_Width / 2 - self.vars.Back_Crotch_Width

    def _step6_waist_distribution(self):
        """步骤6: 腰围分配"""
        d = self.input.design
        self.vars.SS_f = d.Side_Seam_Suppression * 0.4
        self.vars.SS_b = d.Side_Seam_Suppression * 0.6

        self.vars.FW_draft = (self.vars.Net_H / 4 - 1.5 - self.vars.SS_f) - \
            d.Front_Center_Inclination
        self.vars.BW_draft = (self.vars.Net_H / 4 + 1.5 - self.vars.SS_b)
        self.vars.YokeDart = max(0.5, (self.vars.FW_draft + self.vars.BW_draft) - self.vars.Net_W / 2)

    def _step7_knee_ankle(self):
        """步骤7: 膝围/脚口半宽计算"""
        self.vars.Knee_Half_f = (self.vars.Net_Knee / 2) - 1.0
        self.vars.Ankle_Half_f = (self.vars.Net_Ankle / 2) - 1.0
        self.vars.Knee_Half_b = (self.vars.Net_Knee / 2) + 1.0
        self.vars.Ankle_Half_b = (self.vars.Net_Ankle / 2) + 1.0


class InputValidator:
    """输入参数验证器"""

    @staticmethod
    def validate(input_data: JeansPatternInput) -> Tuple[bool, List[str]]:
        """
        验证输入参数
        Returns: (是否通过, 错误信息列表)
        """
        errors = []
        m = input_data.measurements
        d = input_data.design

        # 检查正数
        for name, value in [
            ("W", m.W), ("H", m.H), ("Knee", m.Knee),
            ("Ankle", m.Ankle), ("L", m.L), ("Rise", m.Rise)
        ]:
            if value <= 0:
                errors.append(f"{name} 必须为正数，当前值: {value}")

        # 检查脚口是否过大
        net_ankle = m.Ankle + input_data.ease.Ease_Ankle
        net_knee = m.Knee + input_data.ease.Ease_Knee
        if net_ankle > net_knee + 1.5:
            errors.append(f"脚口围({net_ankle})过大，可能变成微喇裤")

        # 检查臀围线比例
        if not (0.6 <= d.Hip_Line_Ratio <= 0.7):
            errors.append(f"臀围线位置比例必须在[0.6, 0.7]之间")

        # 检查收腰量
        if d.Side_Seam_Suppression > (m.H - m.W) / 2:
            errors.append(f"侧缝收腰量不能超过腰臀差的一半")

        return len(errors) == 0, errors
```

```python
# 文件: src/yoke_rotator.py
"""
育克分割与省量旋转算法
"""
import math
from typing import Tuple, List
from .models import Point, PatternPiece, MaterialType, Grainline, LineSegment


class YokeRotator:
    """育克旋转器"""

    def __init__(self, input_data, vars_data):
        self.input = input_data
        self.vars = vars_data

    def split_and_rotate(self, back_panel: PatternPiece) -> Tuple[PatternPiece, PatternPiece]:
        """
        分割后裤片为育克和后片主片，并旋转省量
        Returns: (yoke, new_back_panel)
        """
        d = self.input.design

        # 步骤1: 生成初始育克分割线
        y1_temp, y2_temp, dividing_points = self._create_dividing_line(back_panel)

        # 步骤2: 找到省尖位置
        dart_tip = self._find_dart_tip(y1_temp, y2_temp)

        # 步骤3: 执行旋转
        yoke_points, back_points = self._rotate_dart(
            dividing_points, dart_tip, back_panel
        )

        # 步骤4: 构建育裁减片
        yoke = self._build_yoke(yoke_points)

        # 步骤5: 构建新后片
        new_back = self._build_back_panel(back_panel, back_points)

        return yoke, new_back

    def _create_dividing_line(self, back_panel: PatternPiece):
        """创建初始育克分割线"""
        d = self.input.design
        points = back_panel.points

        # Y1: 侧缝处向下Yoke_Height
        y1_temp = Point(
            points["B5"].x,
            points["B5"].y + d.Yoke_Height,
            "Y1_temp"
        )

        # Y2: 后中向下Yoke_Height - Yoke_Depth
        y2_temp = Point(
            points["B1"].x,
            points["B1"].y + d.Yoke_Height - d.Yoke_Depth,
            "Y2_temp"
        )

        return y1_temp, y2_temp, [y2_temp, y1_temp]

    def _find_dart_tip(self, y1: Point, y2: Point) -> Point:
        """找到省尖位置：分割线中点向下3cm"""
        mid_x = (y1.x + y2.x) / 2
        mid_y = (y1.y + y2.y) / 2
        return Point(mid_x, mid_y + 3, "dart_tip")

    def _rotate_dart(self, dividing_points, dart_tip: Point, back_panel: PatternPiece):
        """执行仿射旋转"""
        # 计算旋转角度
        # angle = arctan(YokeDart / 省尖到分割线距离)
        distance_to_line = self._distance_point_to_line(
            dart_tip, dividing_points[0], dividing_points[1]
        )
        if distance_to_line < 0.001:
            angle_rad = 0
        else:
            angle_rad = math.atan2(self.vars.YokeDart, distance_to_line)

        # 以省尖为圆心，顺时针旋转
        yoke_points = self._rotate_points_around(
            dividing_points + [back_panel.points["B5"], back_panel.points["B1"]],
            dart_tip, -angle_rad  # 负角度表示顺时针
        )

        return yoke_points, []  # 简化版

    def _rotate_points_around(self, points: List[Point], center: Point, angle_rad: float) -> List[Point]:
        """绕中心点旋转一组点"""
        result = []
        cos_a = math.cos(angle_rad)
        sin_a = math.sin(angle_rad)

        for p in points:
            dx = p.x - center.x
            dy = p.y - center.y
            new_x = center.x + dx * cos_a - dy * sin_a
            new_y = center.y + dx * sin_a + dy * cos_a
            result.append(Point(new_x, new_y, p.name))

        return result

    def _distance_point_to_line(self, p: Point, line_p1: Point, line_p2: Point) -> float:
        """点到直线的距离"""
        return abs((line_p2.y - line_p1.y)*p.x - (line_p2.x - line_p1.x)*p.y +
                   line_p2.x*line_p1.y - line_p2.y*line_p1.x) / \
               math.sqrt((line_p2.y - line_p1.y)**2 + (line_p2.x - line_p1.x)**2)

    def _build_yoke(self, yoke_points) -> PatternPiece:
        """构建育裁减片（简化版）"""
        yoke = PatternPiece(
            name="育克",
            material=MaterialType.SHELL,
            count=2,
            mirror=True
        )
        return yoke

    def _build_back_panel(self, original: PatternPiece, back_points) -> PatternPiece:
        """构建新后片"""
        return original  # 简化版
```

```python
# 文件: src/shrinkage_applier.py
"""
缩水率补偿算法
基于材质进行局部仿射缩放
"""
from .models import (
    Point, PatternPiece, MaterialType, JeansPattern,
    JeansPatternInput
)


class ShrinkageApplier:
    """缩水率补偿器"""

    def __init__(self, input_data: JeansPatternInput):
        self.input = input_data

    def apply(self, pattern: JeansPattern):
        """对所有裁片应用缩水率补偿"""
        pieces = pattern.get_all_pieces()
        for name, piece in pieces:
            if piece:
                self._apply_to_piece(piece)

    def _apply_to_piece(self, piece: PatternPiece):
        """对单个裁片应用缩水率"""
        scale_x, scale_y = self._get_scale_for_material(piece.material)

        if abs(scale_x - 1.0) < 0.001 and abs(scale_y - 1.0) < 0.001:
            return  # 不需要补偿

        # 找到缩放中心（纱向线中点）
        center = self._find_grainline_center(piece)

        # 对所有点进行缩放
        self._scale_points(piece, center, scale_x, scale_y)

    def _get_scale_for_material(self, material: MaterialType) -> tuple:
        """获取材质对应的缩放比例"""
        s = self.input.shrinkage

        if material == MaterialType.SHELL:
            scale_x = 1.0 / (1.0 - s.Shell_Weft / 100.0)
            scale_y = 1.0 / (1.0 - s.Shell_Warp / 100.0)
        elif material == MaterialType.LINING:
            scale_x = 1.0 / (1.0 - s.Lining_Weft / 100.0)
            scale_y = 1.0 / (1.0 - s.Lining_Warp / 100.0)
        else:
            scale_x = 1.0
            scale_y = 1.0

        return scale_x, scale_y

    def _find_grainline_center(self, piece: PatternPiece) -> Point:
        """找到纱向线中点作为缩放中心"""
        if piece.grainline:
            mid_x = (piece.grainline.start.x + piece.grainline.end.x) / 2
            mid_y = (piece.grainline.start.y + piece.grainline.end.y) / 2
            return Point(mid_x, mid_y, "center")

        # 如果没有纱向线，使用所有点的中心
        if piece.net_outline:
            xs = [p.x for p in piece.net_outline]
            ys = [p.y for p in piece.net_outline]
            return Point(sum(xs)/len(xs), sum(ys)/len(ys), "center")

        return Point(0, 0, "center")

    def _scale_points(self, piece: PatternPiece, center: Point, scale_x: float, scale_y: float):
        """对裁片的所有点进行缩放"""
        # 缩放关键点
        for name, p in piece.points.items():
            p.x = center.x + (p.x - center.x) * scale_x
            p.y = center.y + (p.y - center.y) * scale_y

        # 缩放净样轮廓
        if piece.net_outline:
            for p in piece.net_outline:
                p.x = center.x + (p.x - center.x) * scale_x
                p.y = center.y + (p.y - center.y) * scale_y

        # 缩放纱向线
        if piece.grainline:
            for p in [piece.grainline.start, piece.grainline.end]:
                p.x = center.x + (p.x - center.x) * scale_x
                p.y = center.y + (p.y - center.y) * scale_y

        # 缩放剪口
        for notch in piece.notches:
            p = notch.position
            p.x = center.x + (p.x - center.x) * scale_x
            p.y = center.y + (p.y - center.y) * scale_y
```

```python
# 文件: src/seam_allowance.py
"""
缝份添加算法
"""
import math
from typing import List
from .models import Point, PatternPiece, Curve, LineSegment


class SeamRule:
    """缝份规则"""
    def __init__(self, width: float, edge_names: List[str] = None):
        self.width = width
        self.edge_names = edge_names or []


class SeamAllowanceApplier:
    """缝份添加器"""

    def __init__(self):
        # 默认缝份规则
        self.rules = {
            "hem": SeamRule(3.0, ["脚口"]),
            "side": SeamRule(1.0, ["侧缝", "内缝"]),
            "back_crotch": SeamRule(1.5, ["后浪"]),
            "default": SeamRule(1.0)
        }

    def apply(self, piece: PatternPiece, seam_widths: List[float] = None):
        """对裁片添加缝份"""
        if not piece.net_outline:
            return

        if seam_widths is None:
            seam_widths = [1.0] * len(piece.outline_curves)

        piece.seam_widths = seam_widths

        # 计算毛样轮廓
        piece.seam_outline = self._build_seam_outline(
            piece.net_outline, seam_widths
        )

    def _build_seam_outline(self, net_outline: List[Point], seam_widths: List[float]) -> List[Point]:
        """构建毛样轮廓"""
        if len(net_outline) < 2:
            return []

        seam_outline = []

        # 遍历每一段，计算法线方向并偏移
        for i in range(len(net_outline)):
            p_current = net_outline[i]
            p_next = net_outline[(i + 1) % len(net_outline)]
            p_prev = net_outline[(i - 1) % len(net_outline)]

            # 计算两个边的法线
            nx1, ny1 = self._normal(p_prev, p_current)
            nx2, ny2 = self._normal(p_current, p_next)

            # 平均法线（角平分线方向）
            nx = (nx1 + nx2) / 2
            ny = (ny1 + ny2) / 2

            # 归一化
            length = math.hypot(nx, ny)
            if length > 0.001:
                nx /= length
                ny /= length

            # 使用对应边的缝份宽度（简化：统一用第一个）
            width = seam_widths[0] if seam_widths else 1.0

            # 偏移点
            p_seam = Point(
                p_current.x + nx * width,
                p_current.y + ny * width
            )
            seam_outline.append(p_seam)

        return seam_outline

    def _normal(self, p1: Point, p2: Point) -> tuple:
        """计算边的单位法线（向外）"""
        dx = p2.x - p1.x
        dy = p2.y - p1.y

        # 法线：(-dy, dx) 或 (dy, -dx)
        # 假设轮廓是顺时针的，向外法线是 (dy, -dx)
        nx = dy
        ny = -dx

        length = math.hypot(nx, ny)
        if length > 0.001:
            nx /= length
            ny /= length

        return nx, ny
```

```python
        self.scale_factor = self._get_scale_factor()

    def _get_scale_factor(self) -> float:
        """获取单位转换系数（转成mm）"""
        if self.units_scale == "mm":
            return 1.0
        elif self.units_scale == "cm":
            return 10.0
        elif self.units_scale == "m":
            return 1000.0
        elif self.units_scale == "inch":
            return 25.4
        return 10.0

    def export(self, pattern: JeansPattern, filepath: str, separate_pieces: bool = False):
        """
        导出DXF文件
        separate_pieces: True=每个裁片一个文件，False=所有裁片一个文件
        """
        if separate_pieces:
            self._export_separate(pattern, filepath)
        else:
            self._export_single(pattern, filepath)

    def _export_single(self, pattern: JeansPattern, filepath: str):
        """所有裁片导出到一个文件"""
        doc = ezdxf.new("R12")
        doc.units = units.MM
        msp = doc.modelspace()

        self._setup_layers(doc)

        # 逐个排列裁片
        offset_x = 0.0
        spacing = 50.0  # 裁片间距5cm=50mm

        pieces = pattern.get_all_pieces()
        for name, piece in pieces:
            if piece:
                self._draw_piece(msp, piece, offset_x, 0.0)
                offset_x += self._estimate_width(piece) + spacing

        doc.saveas(filepath)

    def _export_separate(self, pattern: JeansPattern, filepath_pattern: str):
        """每个裁片单独一个文件"""
        pieces = pattern.get_all_pieces()
        for name, piece in pieces:
            if piece:
                doc = ezdxf.new("R12")
                doc.units = units.MM
                msp = doc.modelspace()
                self._setup_layers(doc)
                self._draw_piece(msp, piece, 0, 0)

                filepath = filepath_pattern.replace("{name}", name)
                doc.saveas(filepath)

    def _setup_layers(self, doc):
        """设置AAMA标准图层"""
        for layer in DXFLayer:
            doc.layers.new(name=layer.name, dxfattribs={"color": layer.value})

    def _draw_piece(self, msp, piece: PatternPiece, offset_x: float, offset_y: float):
        """绘制单个裁片"""
        scale = self.scale_factor

        # 1. 净样线 (NET层)
        if piece.net_outline:
            points = [
                (p.x * scale + offset_x, p.y * scale + offset_y)
                for p in piece.net_outline
            ]
            # 闭合
            if points and points[0] != points[-1]:
                points.append(points[0])
            msp.add_lwpolyline(points, dxfattribs={"layer": "NET"})

        # 2. 毛样线 (SEAM层)
        if piece.seam_outline:
            points = [
                (p.x * scale + offset_x, p.y * scale + offset_y)
                for p in piece.seam_outline
            ]
            if points and points[0] != points[-1]:
                points.append(points[0])
            msp.add_lwpolyline(points, dxfattribs={"layer": "SEAM"})

        # 3. 纱向线 (GRAIN层)
        if piece.grainline:
            start = (
                piece.grainline.start.x * scale + offset_x,
                piece.grainline.start.y * scale + offset_y
            )
            end = (
                piece.grainline.end.x * scale + offset_x,
                piece.grainline.end.y * scale + offset_y
            )
            msp.add_line(start, end, dxfattribs={"layer": "GRAIN"})

            # 添加箭头
            self._draw_arrow(msp, start, end, "GRAIN")

        # 4. 剪口 (NOTCH层)
        for notch in piece.notches:
            self._draw_notch(msp, notch, offset_x, offset_y, scale)

        # 5. 钻孔点 (DRILL层)
        for hole in piece.drill_holes:
            center = (hole.x * scale + offset_x, hole.y * scale + offset_y)
            msp.add_circle(center, 1.0, dxfattribs={"layer": "DRILL"})

        # 6. 文字标注 (TEXT层)
        text_pos = self._get_text_position(piece, offset_x, offset_y, scale)
        text = f"{piece.name} x{piece.count}"
        if piece.mirror:
            text += " (需镜像)"
        msp.add_text(
            text,
            height=5.0,
            dxfattribs={"layer": "TEXT"}
        ).set_pos(text_pos, align="MIDDLE_CENTER")

    def _draw_arrow(self, msp, start: tuple, end: tuple, layer: str):
        """绘制箭头"""
        dx = end[0] - start[0]
        dy = end[1] - start[1]
        length = (dx**2 + dy**2)**0.5
        if length < 0.1:
            return

        # 单位向量
        ux = dx / length
        uy = dy / length

        # 箭头大小
        arrow_len = 5.0
        arrow_angle = 3.14159 * 0.25  # 45度

        # 末端箭头
        cos_a = arrow_len * 0.7
        sin_a = arrow_len * 0.7
        p1 = (end[0] - ux * cos_a + uy * sin_a,
              end[1] - uy * cos_a - ux * sin_a)
        p2 = (end[0] - ux * cos_a - uy * sin_a,
              end[1] - uy * cos_a + ux * sin_a)

        msp.add_line(end, p1, dxfattribs={"layer": layer})
        msp.add_line(end, p2, dxfattribs={"layer": layer})

        # 起始箭头
        p1 = (start[0] + ux * cos_a + uy * sin_a,
              start[1] + uy * cos_a - ux * sin_a)
        p2 = (start[0] + ux * cos_a - uy * sin_a,
              start[1] + uy * cos_a + ux * sin_a)

        msp.add_line(start, p1, dxfattribs={"layer": layer})
        msp.add_line(start, p2, dxfattribs={"layer": layer})

    def _draw_notch(self, msp, notch, offset_x, offset_y, scale):
        """绘制剪口（简化的I型）"""
        p = notch.position
        x = p.x * scale + offset_x
        y = p.y * scale + offset_y
        depth = notch.depth * scale

        msp.add_line((x - depth, y), (x + depth, y), dxfattribs={"layer": "NOTCH"})

    def _get_text_position(self, piece: PatternPiece, offset_x, offset_y, scale) -> tuple:
        """获取文字标注位置"""
        if piece.net_outline:
            xs = [p.x for p in piece.net_outline]
            ys = [p.y for p in piece.net_outline]
            cx = (min(xs) + max(xs)) / 2 * scale + offset_x
            cy = (min(ys) + max(ys)) / 2 * scale + offset_y
            return (cx, cy)
        return (offset_x, offset_y)

    def _estimate_width(self, piece: PatternPiece) -> float:
        """估算裁片宽度（用于排列）"""
        if piece.net_outline:
            xs = [p.x for p in piece.net_outline]
            return (max(xs) - min(xs)) * self.scale_factor
        return 100.0
```

```python
# 文件: src/jeans_system.py
"""
牛仔裤打版系统主入口类
"""
from typing import Optional, Tuple
from .models import (
    JeansPatternInput, Measurements, EaseParameters,
    ShrinkageParameters, DesignParameters, JeansPattern
)
from .calculator import InputValidator, IntermediateCalculator
from .pattern_generator import FrontPanelGenerator, BackPanelGenerator
from .yoke_rotator import YokeRotator
from .validator import SeamValidator
from .shrinkage_applier import ShrinkageApplier
from .seam_allowance import SeamAllowanceApplier
from .dxf_exporter import DXFExporter


class JeansPatternSystem:
    """
    牛仔裤打版系统

    使用示例:
    >>> system = JeansPatternSystem()
    >>> measurements = Measurements(W=70, H=96, Knee=42, Ankle=34, L=102, Rise=28, FCL=24, BCL=30)
    >>> pattern = system.generate(measurements)
    >>> system.export_dxf(pattern, "jeans.dxf")
    """

    def __init__(self):
        pass

    def generate(
        self,
        measurements: Measurements,
        ease: Optional[EaseParameters] = None,
        shrinkage: Optional[ShrinkageParameters] = None,
        design: Optional[DesignParameters] = None
    ) -> JeansPattern:
        """
        生成完整牛仔裤版型

        Args:
            measurements: 净尺寸
            ease: 松量参数（默认使用预设值）
            shrinkage: 缩水率参数（默认使用预设值）
            design: 设计参数（默认使用预设值）

        Returns:
            JeansPattern: 完整版型对象
        """
        # 构建输入参数
        input_data = JeansPatternInput(
            measurements=measurements,
            ease=ease or EaseParameters(),
            shrinkage=shrinkage or ShrinkageParameters(),
            design=design or DesignParameters()
        )

        # 步骤1: 验证输入
        valid, errors = InputValidator.validate(input_data)
        if not valid:
            raise ValueError(f"输入参数验证失败: {errors}")

        # 步骤2: 计算中间变量
        calculator = IntermediateCalculator(input_data)
        vars_data = calculator.calculate()

        # 步骤3: 创建版型对象
        pattern = JeansPattern(
            input=input_data,
            intermediate=vars_data
        )

        # 步骤4: 生成前裤片
        front_gen = FrontPanelGenerator(input_data, vars_data)
        pattern.front_panel = front_gen.generate()

        # 步骤5: 生成后裤片
        back_gen = BackPanelGenerator(input_data, vars_data)
        pattern.back_panel = back_gen.generate()

        # 步骤6: 育克分割与旋转
        yoke_rotator = YokeRotator(input_data, vars_data)
        pattern.yoke, pattern.back_panel = yoke_rotator.split_and_rotate(pattern.back_panel)

        # 步骤7: 生成零部件（简化版）
        self._generate_components(pattern, input_data, vars_data)

        # 步骤8: 接缝校验
        validator = SeamValidator()
        validator.validate(pattern)

        # 步骤9: 缩水率补偿
        shrinkage_applier = ShrinkageApplier(input_data)
        shrinkage_applier.apply(pattern)

        # 步骤10: 添加缝份
        seam_applier = SeamAllowanceApplier()
        for name, piece in pattern.get_all_pieces():
            if piece:
                seam_applier.apply(piece)

        return pattern

    def _generate_components(self, pattern: JeansPattern, input_data, vars_data):
        """生成零部件（简化版占位）"""
        from .models import PatternPiece, MaterialType

        # 腰头（占位）
        pattern.waistband = PatternPiece(
            name="弯腰头", material=MaterialType.SHELL,
            count=1, mirror=False, unfold_required=True
        )

        # 后贴袋（占位）
        pattern.back_pocket = PatternPiece(
            name="后贴袋", material=MaterialType.SHELL,
            count=2, mirror=True
        )

        # 前袋贴（占位）
        pattern.front_pocket_facing = PatternPiece(
            name="前袋贴", material=MaterialType.SHELL,
            count=2, mirror=True
        )

        # 表袋（占位）
        pattern.coin_pocket = PatternPiece(
            name="表袋", material=MaterialType.SHELL,
            count=2, mirror=True
        )

        # 前袋布（占位）
        pattern.front_pocket_bag = PatternPiece(
            name="前袋布", material=MaterialType.LINING,
            count=2, mirror=True
        )

        # 门襟（占位）
        pattern.fly_facing = PatternPiece(
            name="门襟", material=MaterialType.SHELL,
            count=1, mirror=False
        )

        # 里襟（占位）
        pattern.fly_shield = PatternPiece(
            name="里襟", material=MaterialType.SHELL,
            count=1, mirror=False
        )

        # 串带袢（占位）
        pattern.belt_loop = PatternPiece(
            name="串带袢", material=MaterialType.SHELL,
            count=7, mirror=False
        )

    def export_dxf(self, pattern: JeansPattern, filepath: str):
        """导出DXF文件"""
        exporter = DXFExporter(units_scale="cm")
        exporter.export(pattern, filepath)
```

## 10. 注意事项

### 10.1 常见问题

1. **坐标系方向**：Y轴向下，符合服装制版惯例
2. **单位**：建议统一使用cm，DXF输出时自动转mm
3. **容差**：曲线长度校验容差0.1cm
4. **后内缝**：后内缝比前内缝短1cm，用于缝制时拉伸
5. **剪口深度**：不超过缝份宽度的一半
